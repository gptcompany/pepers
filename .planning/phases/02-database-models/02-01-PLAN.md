```sql
-- schema_version table for migration tracking
CREATE TABLE IF NOT EXISTS schema_version (
    version INTEGER PRIMARY KEY,

-- papers: academic paper metadata (Discovery service)
CREATE TABLE IF NOT EXISTS papers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    arxiv_id TEXT UNIQUE NOT NULL,
    title TEXT NOT NULL,
    abstract TEXT,
    authors TEXT,              -- JSON array of author names
    categories TEXT,           -- JSON array of arXiv categories
    doi TEXT,
    pdf_url TEXT,
    published_date TEXT,       -- ISO 8601
    -- Semantic Scholar enrichment
    semantic_scholar_id TEXT,
    citation_count INTEGER DEFAULT 0,
    reference_count INTEGER DEFAULT 0,
    influential_citation_count INTEGER DEFAULT 0,
    venue TEXT,
    fields_of_study TEXT,      -- JSON array
    tldr TEXT,
    open_access INTEGER DEFAULT 0,  -- boolean
    -- CrossRef enrichment
    crossref_data TEXT,        -- JSON blob
    -- Pipeline state
    stage TEXT NOT NULL DEFAULT 'discovered',
    score REAL,                -- analyzer relevance score (0-50)
    error TEXT,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);

-- formulas: extracted LaTeX formulas (Extractor service)
CREATE TABLE IF NOT EXISTS formulas (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    paper_id INTEGER NOT NULL REFERENCES papers(id),
    latex TEXT NOT NULL,
    latex_hash TEXT NOT NULL,      -- SHA-256 of raw latex for dedup
    description TEXT,              -- LLM plain-language description
    formula_type TEXT,             -- 'inline', 'display', 'equation'
    context TEXT,                  -- surrounding text from paper
    stage TEXT NOT NULL DEFAULT 'extracted',
    error TEXT,
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);

-- validations: CAS validation results (Validator service)
CREATE TABLE IF NOT EXISTS validations (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    formula_id INTEGER NOT NULL REFERENCES formulas(id),
    engine TEXT NOT NULL,          -- 'sympy', 'maxima', 'wolfram'
    is_valid INTEGER,             -- boolean: 1=valid, 0=invalid, NULL=error
    result TEXT,                   -- engine output
    error TEXT,
    time_ms INTEGER,
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);

-- generated_code: code generation results (Codegen service)
CREATE TABLE IF NOT EXISTS generated_code (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    formula_id INTEGER NOT NULL REFERENCES formulas(id),
    language TEXT NOT NULL,        -- 'python', 'rust', future languages
    code TEXT NOT NULL,
    metadata TEXT,                 -- JSON blob (model used, compile status, etc.)
    stage TEXT NOT NULL DEFAULT 'codegen',
    error TEXT,
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);
```

**get_connection(db_path)** implementation:
1. Ensure parent directory exists (`Path(db_path).parent.mkdir(parents=True, exist_ok=True)`)
2. Connect with `sqlite3.connect(str(db_path))`
3. Set pragmas: `journal_mode=WAL`, `foreign_keys=ON`
4. Set `row_factory = sqlite3.Row`
5. Return connection

**transaction(db_path)** implementation:
1. Get connection via `get_connection(db_path)`
2. Try: yield connection, then `conn.commit()`
3. Except: `conn.rollback()`, then re-raise
4. Finally: `conn.close()`

**init_db(db_path)** implementation:
1. Get connection via `get_connection(db_path)`
2. Execute SCHEMA (each CREATE TABLE statement)
3. Execute INDEX creation
4. Insert schema_version row (version=1) if not exists: `INSERT OR IGNORE INTO schema_version (version) VALUES (1)`
5. Commit and close

Keep imports minimal: sqlite3, contextlib, pathlib, typing. No external deps.

import sys, tempfile, os; sys.path.insert(0, '.')
from shared.db import init_db, transaction
from shared.models import Paper, Formula, Validation, GeneratedCode, PipelineStage

db = os.path.join(tempfile.mkdtemp(), 'test.db')
init_db(db)

# Insert paper
with transaction(db) as conn:
    p = Paper(arxiv_id='2401.00001', title='Kelly Criterion Revisited', authors=['J. Smith'], categories=['q-fin.PM'])
    d = p.model_dump(exclude={'id', 'created_at', 'updated_at'})
    # Serialize list/dict fields to JSON strings for SQLite
    import json
    for k, v in d.items():
        if isinstance(v, (list, dict)):
            d[k] = json.dumps(v)
        elif isinstance(v, bool):
            d[k] = int(v)
        elif hasattr(v, 'value'):  # Enum
            d[k] = v.value
    cols = ', '.join(d.keys())
    placeholders = ', '.join(':' + k for k in d.keys())
    conn.execute(f'INSERT INTO papers ({cols}) VALUES ({placeholders})', d)
    row = conn.execute('SELECT * FROM papers WHERE arxiv_id = ?', ('2401.00001',)).fetchone()
    p2 = Paper.model_validate(dict(row))
    assert p2.title == 'Kelly Criterion Revisited'
    assert p2.authors == ['J. Smith']
    paper_id = row['id']

# Insert formula
with transaction(db) as conn:
