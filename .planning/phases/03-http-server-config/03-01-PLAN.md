```python
class JsonFormatter(logging.Formatter):
    """JSON log formatter for Loki/journald."""

    def __init__(self, service: str = "unknown"):
        super().__init__()
        self.service = service

    def format(self, record: logging.LogRecord) -> str:
        log_entry = {
            "timestamp": self.formatTime(record, "%Y-%m-%dT%H:%M:%S"),
            "level": record.levelname,
            "service": self.service,
            "msg": record.getMessage(),
        }
        if record.exc_info and record.exc_info[0]:
            log_entry["exception"] = self.formatException(record.exc_info)
        return json.dumps(log_entry)
```

**2. route() decorator:**

```python
def route(method: str, path: str) -> Callable:
    def decorator(fn: Callable) -> Callable:
        fn._route = (method.upper(), path)
        return fn
    return decorator
```

**3. BaseHandler — full implementation:**

```python
class BaseHandler(BaseHTTPRequestHandler):
    # Class-level registry built on first request
    _routes: dict[tuple[str, str], str] | None = None

    @classmethod
    def _build_routes(cls) -> dict[tuple[str, str], str]:
        """Scan class methods for @route decorators."""
        routes = {}
        for name in dir(cls):
            method = getattr(cls, name, None)
            if callable(method) and hasattr(method, "_route"):
                routes[method._route] = name
        return routes

    def _dispatch(self, http_method: str) -> None:
        """Dispatch request to matching route handler."""
        if self.__class__._routes is None:
            self.__class__._routes = self._build_routes()
        key = (http_method, self.path)
        # Also try path without query string
        path_no_query = self.path.split("?")[0]
        handler_name = self._routes.get(key) or self._routes.get((http_method, path_no_query))
        if handler_name:
            handler = getattr(self, handler_name)
            try:
                if http_method == "POST":
                    data = self.read_json()
                    if data is None:
                        return  # error already sent
                    result = handler(data)
                else:
                    result = handler()
                if result is not None:
                    self.send_json(result)
            except Exception as e:
                logger.exception("Handler error: %s", e)
                self.send_error_json(str(e), "INTERNAL_ERROR", 500)
        else:
            self.send_error_json(f"Not found: {self.path}", "NOT_FOUND", 404)

    def do_GET(self) -> None:
        self._dispatch("GET")

    def do_POST(self) -> None:
        self._dispatch("POST")

    def send_json(self, data: dict | list, status: int = 200) -> None:
        body = json.dumps(data, default=str).encode("utf-8")
        self.send_response(status)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", str(len(body)))
        self.end_headers()
        self.wfile.write(body)

    def send_error_json(self, error: str, code: str, status: int = 400,
                        details: dict | None = None) -> None:
        body: dict[str, Any] = {"error": error, "code": code}
        if details:
            body["details"] = details
        self.send_json(body, status)

    def read_json(self) -> dict | None:
        content_length = int(self.headers.get("Content-Length", 0))
        if content_length == 0:
            self.send_error_json("Request body is empty", "INVALID_JSON", 400)
            return None
        try:
            body = self.rfile.read(content_length)
            return json.loads(body)
        except (json.JSONDecodeError, ValueError) as e:
            self.send_error_json(f"Invalid JSON: {e}", "INVALID_JSON", 400)
            return None

    def log_message(self, format: str, *args: Any) -> None:
        """Override default log to use Python logging."""
        logger.info("%s %s", self.client_address[0], format % args)
```

**4. BaseService — full implementation:**

```python
class BaseService:
    def __init__(self, name: str, port: int, handler: type[BaseHandler],
                 db_path: str | None = None) -> None:
        self.name = name
        self.port = port
        self.handler = handler
        self.db_path = db_path
        self.server: HTTPServer | None = None
        self._start_time = time.time()

        # Inject service metadata into handler class
        handler.service_name = name
        handler.service_version = "0.1.0"
        handler.service_start_time = self._start_time
        handler.db_path = db_path

        # Register built-in routes
        self._register_builtins(handler)

    def _register_builtins(self, handler: type[BaseHandler]) -> None:
        """Register /health and /status as built-in routes."""

        @route("GET", "/health")
        def handle_health(self_handler):
            return {
                "status": "ok",
                "service": handler.service_name,
                "uptime_seconds": round(time.time() - handler.service_start_time, 1),
            }

        @route("GET", "/status")
        def handle_status(self_handler):
            result = {
                "service": handler.service_name,
                "version": handler.service_version,
                "uptime_seconds": round(time.time() - handler.service_start_time, 1),
            }
            if handler.db_path:
                result["db_path"] = handler.db_path
            return result

        # Bind to handler class only if not already defined
        if not hasattr(handler, "handle_health") or not hasattr(getattr(handler, "handle_health", None), "_route"):
            handler.handle_health = handle_health
        if not hasattr(handler, "handle_status") or not hasattr(getattr(handler, "handle_status", None), "_route"):
            handler.handle_status = handle_status

        # Reset route cache
        handler._routes = None

    def _setup_logging(self) -> None:
        """Configure JSON structured logging."""
        root = logging.getLogger()
        root.setLevel(logging.INFO)
        # Remove existing handlers
        root.handlers.clear()
        # Add JSON handler to stderr (for journald)
        handler = logging.StreamHandler()
        handler.setFormatter(JsonFormatter(service=self.name))
        root.addHandler(handler)

    def _handle_sigterm(self, signum: int, frame: Any) -> None:
        """Handle SIGTERM for graceful shutdown."""
        logger.info("SIGTERM received, shutting down...")
        if self.server:
            # shutdown() waits for current request to finish
            self.server.shutdown()

    def run(self) -> None:
        self._setup_logging()
        signal.signal(signal.SIGTERM, self._handle_sigterm)
        self.server = HTTPServer(("0.0.0.0", self.port), self.handler)
        logger.info("Starting %s on port %d", self.name, self.port)
        try:
            self.server.serve_forever()
        except KeyboardInterrupt:
            pass
        finally:
            if self.server:
                self.server.server_close()
            logger.info("%s stopped", self.name)
```

**IMPORTANT NOTES:**
- The `_register_builtins` approach injects health/status as methods on the handler class. This way they go through the normal route dispatch.
- `JsonFormatter` outputs one JSON line per log entry, which Promtail/Loki parses natively.
- `send_json` uses `default=str` in json.dumps to handle datetime objects without crashing.
- `_dispatch` strips query strings from path for matching.
- `log_message` override prevents default http.server access log spam, routing it through Python logging instead.
